\keys_define:nn { cover/page }
  {
    date .tl_set:N = \l_cover_date_tl,
    title .tl_set:N = \l_cover_title_tl,
    subtitle .tl_set:N = \l_cover_subtitle_tl,
    committee .tl_set:N = \l_cover_committee_tl,
    title_max_lines .int_set:N = \l_cover_title_max_lines_int,
    title_fontsize .tl_set:N = \l_cover_title_fontsize_tl,
    title_baselineskip .tl_set:N = \l_cover_title_baselineskip_tl,
  }

\cs_new_protected:Npn \cover_check_title_fit:
  {
    \vbox_set:Nn \l_cover_title_measure_box
      {
        \hsize=175.741mm
        \parindent=0pt
        \centering
        \bfseries\fontsize{\l_cover_title_fontsize_tl}{\l_cover_title_baselineskip_tl}\selectfont
        \tl_use:N \l_cover_title_tl \par
      }
    \dim_set:Nn \l_cover_title_used_dim
      { \box_ht:N \l_cover_title_measure_box + \box_dp:N \l_cover_title_measure_box }
    \dim_zero:N \l_cover_title_allowed_dim
    \int_step_inline:nn { \l_cover_title_max_lines_int }
      { \dim_add:Nn \l_cover_title_allowed_dim { \l_cover_title_baselineskip_tl } }
    \dim_compare:nNnT { \l_cover_title_used_dim } > { \l_cover_title_allowed_dim }
      {
        \PackageWarningNoLine{PageDeGardeIHEDN}
          {Title~height~(\dim_use:N \l_cover_title_used_dim)~exceeds~title_max_lines~
           (\int_use:N\l_cover_title_max_lines_int).~
           Consider~reducing~title_fontsize/title_baselineskip~or~shortening~title}
      }
  }

\cs_new_protected:Npn \cover_set_fields:n #1
  {
    \tl_clear:N \l_cover_date_tl
    \tl_clear:N \l_cover_title_tl
    \tl_clear:N \l_cover_subtitle_tl
    \tl_clear:N \l_cover_committee_tl
    \tl_set:Nn \l_cover_title_fontsize_tl {26.000pt}
    \tl_set:Nn \l_cover_title_baselineskip_tl {28.667pt}
    \int_set:Nn \l_cover_title_max_lines_int {2}
    \keys_set:nn { cover/page } { #1 }
    \tl_trim_spaces:N \l_cover_committee_tl
    \tl_replace_all:Nnn \l_cover_committee_tl { \\ } { \c_space_tl }
    \tl_replace_all:Nnn \l_cover_committee_tl { \par } { \par\noindent }
    \cover_check_title_fit:
  }
